## Sistema de Roles Implementado

### Roles Definidos
- **PARTICIPANTE**: Usuarios que pueden participar en concursos
- **JURADO**: Usuarios que pueden evaluar medios asignados
- **ADMIN**: Administradores con acceso completo al sistema
- **CONTENT_ADMIN**: Administradores específicos para gestión de contenido CMS

### Características del Sistema
- **Autorización granular** por endpoint y recurso
- **Middleware especializado** para cada tipo de rol
- **Verificación de propiedad** de recursos por usuario
- **Guards flexibles** para combinaciones de roles

### 1. Middleware de Autorización por Roles

#### Sistema Principal de Roles
```typescript
// src/middleware/role.middleware.ts
import { Response, NextFunction } from 'express';
import { AuthenticatedRequest } from './auth.middleware';

export type UserRole = 'PARTICIPANTE' | 'JURADO' | 'ADMIN' | 'CONTENT_ADMIN';

export function requireRole(allowedRoles: UserRole[]) {
  return (req: AuthenticatedRequest, res: Response, next: NextFunction): void => {
    if (!req.user) {
      res.status(401).json({
        success: false,
        error: 'Autenticación requerida',
        code: 'AUTHENTICATION_REQUIRED'
      });
      return;
    }

    const userRole = req.user.role as UserRole;

    if (!allowedRoles.includes(userRole)) {
      res.status(403).json({
        success: false,
        error: 'Permisos insuficientes',
        code: 'INSUFFICIENT_PERMISSIONS',
        required_roles: allowedRoles,
        user_role: userRole
      });
      return;
    }

    next();
  };
}
```

```typescript
export function requireParticipante(req: AuthenticatedRequest, res: Response, next: NextFunction): void {
  requireRole(['PARTICIPANTE', 'JURADO', 'ADMIN', 'CONTENT_ADMIN'])(req, res, next);
}
```

#### Verificación de Propiedad de Recursos
```typescript
// Middleware para verificar propiedad de recurso
export function requireOwnershipOrAdmin(getUserIdFromResource: (req: AuthenticatedRequest) => string) {
  return (req: AuthenticatedRequest, res: Response, next: NextFunction): void => {
    if (!req.user) {
      res.status(401).json({
        success: false,
        error: 'Autenticación requerida'
      });
      return;
    }

    const userRole = req.user.role as UserRole;
    const resourceUserId = getUserIdFromResource(req);

    // Admins pueden acceder a cualquier recurso
    if (userRole === 'ADMIN') {
      next();
      return;
    }

    // El usuario debe ser el propietario del recurso
    if (req.user.userId !== resourceUserId) {
      res.status(403).json({
        success: false,
        error: 'Solo puedes acceder a tus propios recursos',
        code: 'RESOURCE_OWNERSHIP_REQUIRED'
      });
      return;
    }

    next();
  };
}

// Middleware para verificar propiedad de medio
export const requireMediaOwnership = requireOwnershipOrAdmin((req) => {
  // Extraer usuario_id del medio desde parámetros o body
  return req.body.usuario_id || req.params.userId || '';
});

// Middleware para verificar propiedad de calificación
export const requireCalificacionOwnership = requireOwnershipOrAdmin((req) => {
  // Para calificaciones, el jurado debe ser el propietario
  return req.user?.userId || '';
});
```

### 2. Guards Avanzados para Casos Específicos

#### Guard para Jurados con Asignación
```typescript
// src/middleware/jurado-assignment.middleware.ts
import { Response, NextFunction } from 'express';
import { AuthenticatedRequest } from './auth.middleware';
import { prisma } from '../lib/prisma';

export function requireJuradoAssignment(getCategoriaId: (req: AuthenticatedRequest) => number) {
  return async (req: AuthenticatedRequest, res: Response, next: NextFunction): Promise<void> => {
    if (!req.user) {
      res.status(401).json({
        success: false,
        error: 'Autenticación requerida'
      });
      return;
    }

    const userRole = req.user.role as UserRole;
    const userId = req.user.userId;

    // Admins pueden acceder sin asignación específica
    if (userRole === 'ADMIN') {
      next();
      return;
    }

    // Verificar que el usuario sea jurado
    if (userRole !== 'JURADO') {
      res.status(403).json({
        success: false,
        error: 'Se requiere rol de jurado',
        code: 'JURADO_ROLE_REQUIRED'
      });
      return;
    }

    try {
      const categoriaId = getCategoriaId(req);
      
      // Verificar asignación del jurado a la categoría
      const asignacion = await prisma.juradoAsignacion.findUnique({
        where: {
          usuario_id_categoria_id: {
            usuario_id: userId,
            categoria_id: categoriaId
          }
        }
      });

      if (!asignacion) {
        res.status(403).json({
          success: false,
          error: 'No tienes asignación para evaluar esta categoría',
          code: 'CATEGORY_ASSIGNMENT_REQUIRED'
        });
        return;
      }

      next();
    } catch (error) {
      console.error('Error verificando asignación de jurado:', error);
      res.status(500).json({
        success: false,
        error: 'Error interno del servidor'
      });
    }
  };
}
```

#### Guard para Contenido CMS
```typescript
// src/middleware/cms-permissions.middleware.ts
import { Response, NextFunction } from 'express';
import { AuthenticatedRequest } from './auth.middleware';
import { TipoContenido } from '@prisma/client';

export function requireCMSPermissions(allowedContentTypes?: TipoContenido[]) {
  return (req: AuthenticatedRequest, res: Response, next: NextFunction): void => {
    if (!req.user) {
      res.status(401).json({
        success: false,
        error: 'Autenticación requerida'
      });
      return;
    }

    const userRole = req.user.role as UserRole;

    // Admins tienen acceso completo
    if (userRole === 'ADMIN') {
      next();
      return;
    }

    // Content admins tienen acceso a contenido CMS
    if (userRole === 'CONTENT_ADMIN') {
      // Si se especifican tipos de contenido permitidos, validar
      if (allowedContentTypes && req.body.tipo_contenido) {
        const tipoContenido = req.body.tipo_contenido as TipoContenido;
        
        if (!allowedContentTypes.includes(tipoContenido)) {
          res.status(403).json({
            success: false,
            error: 'No tienes permisos para este tipo de contenido',
            code: 'CONTENT_TYPE_NOT_ALLOWED',
            allowed_types: allowedContentTypes
          });
          return;
        }
      }
      
      next();
      return;
    }

    res.status(403).json({
      success: false,
      error: 'Se requieren permisos de administrador de contenido',
      code: 'CMS_PERMISSIONS_REQUIRED'
    });
  };
}

// Guard específico para páginas estáticas (solo ADMIN)
export const requireStaticPagePermissions = requireRole(['ADMIN']);

// Guard para contenido educativo (CONTENT_ADMIN y ADMIN)
export const requireEducationalContentPermissions = requireCMSPermissions([
  TipoContenido.CONTENIDO_EDUCATIVO,
  TipoContenido.BLOG_POST
]);
```

```typescript
// src/routes/concurso.routes.ts
import { Router } from 'express';
import { concursoController } from '../controllers/concurso.controller';
import { authenticateToken } from '../middleware/auth.middleware';
import { requireRole, requireAdmin } from '../middleware/role.middleware';

const router = Router();

// Rutas públicas (sin autenticación)
router.get('/activos', concursoController.getConcursosActivos);
router.get('/finalizados', concursoController.getConcursosFinalizados);
router.get('/:id', concursoController.getConcursoById);

// Rutas para usuarios autenticados
router.use(authenticateToken);

// Rutas para participantes (cualquier usuario autenticado)
router.post('/inscripcion', requireRole(['PARTICIPANTE', 'JURADO', 'ADMIN']), concursoController.inscribirseAConcurso);
router.delete('/inscripcion/:concursoId', requireRole(['PARTICIPANTE', 'JURADO', 'ADMIN']), concursoController.cancelarInscripcion);
router.get('/mis-inscripciones', requireRole(['PARTICIPANTE', 'JURADO', 'ADMIN']), concursoController.getMisInscripciones);

// Rutas solo para administradores
router.get('/', requireAdmin, concursoController.getConcursos);
router.post('/', requireAdmin, concursoController.createConcurso);
router.put('/:id', requireAdmin, concursoController.updateConcurso);
router.delete('/:id', requireAdmin, concursoController.deleteConcurso);

export default router;
```

### 4. Validaciones Específicas por Rol

#### Validación en Controladores
```typescript
// src/controllers/user.controller.ts
export class UserController {
  /**
   * Actualizar perfil de usuario
   * Los usuarios solo pueden actualizar su propio perfil, excepto admins
   */
  async updateProfile(req: AuthenticatedRequest, res: Response): Promise<void> {
    try {
      const targetUserId = req.params.userId;
      const currentUserId = req.user!.userId;
      const currentUserRole = req.user!.role as UserRole;

      // Verificar permisos
      if (currentUserRole !== 'ADMIN' && targetUserId !== currentUserId) {
        res.status(403).json({
          success: false,
          error: 'Solo puedes actualizar tu propio perfil',
          code: 'PROFILE_OWNERSHIP_REQUIRED'
        });
        return;
      }

      // Validar que no se intente cambiar el rol sin ser admin
      if (req.body.role && currentUserRole !== 'ADMIN') {
        res.status(403).json({
          success: false,
          error: 'Solo los administradores pueden cambiar roles',
          code: 'ROLE_CHANGE_FORBIDDEN'
        });
        return;
      }

      const updatedUser = await userService.updateProfile(targetUserId, req.body);

      res.json({
        success: true,
        data: updatedUser,
        message: 'Perfil actualizado exitosamente'
      });
    } catch (error) {
      console.error('Error al actualizar perfil:', error);
      res.status(500).json({
        success: false,
        error: 'Error interno del servidor'
      });
    }
  }

  /**
   * Cambiar rol de usuario (solo admins)
   */
  async changeUserRole(req: AuthenticatedRequest, res: Response): Promise<void> {
    try {
      // Esta ruta ya está protegida por requireAdmin middleware
      const targetUserId = req.params.userId;
      const newRole = req.body.role as UserRole;

      // Validar que el rol sea válido
      const validRoles: UserRole[] = ['PARTICIPANTE', 'JURADO', 'ADMIN', 'CONTENT_ADMIN'];
      if (!validRoles.includes(newRole)) {
        res.status(400).json({
          success: false,
          error: 'Rol inválido',
          valid_roles: validRoles
        });
        return;
      }

      const updatedUser = await userService.changeRole(targetUserId, newRole);

      res.json({
        success: true,
        data: updatedUser,
        message: `Rol cambiado a ${newRole} exitosamente`
      });
    } catch (error) {
      console.error('Error al cambiar rol:', error);
      res.status(500).json({
        success: false,
        error: 'Error interno del servidor'
      });
    }
  }
}
```
