### Configuración de Conexión
```typescript
// src/lib/prisma.ts
import { PrismaClient } from '@prisma/client';

const globalForPrisma = globalThis as unknown as {
  prisma: PrismaClient | undefined;
};

export const prisma = globalForPrisma.prisma ?? new PrismaClient({
  log: process.env.NODE_ENV === 'development' ? ['query', 'error', 'warn'] : ['error'],
  datasources: {
    db: {
      url: process.env.DATABASE_URL,
    },
  },
});

if (process.env.NODE_ENV !== 'production') globalForPrisma.prisma = prisma;

// Función para conectar a la base de datos
export async function connectDatabase() {
  try {
    await prisma.$connect();
    console.log('✅ Conectado a PostgreSQL exitosamente');
    
    // Verificar conexión con una consulta simple
    await prisma.$queryRaw`SELECT 1`;
    console.log('✅ Verificación de conexión exitosa');
  } catch (error) {
    console.error('❌ Error conectando a la base de datos:', error);
    throw error;
  }
}

// Función para desconectar
export async function disconnectDatabase() {
  try {
    await prisma.$disconnect();
    console.log('✅ Desconectado de PostgreSQL');
  } catch (error) {
    console.error('❌ Error desconectando de la base de datos:', error);
  }
}
```

```typescript
  }
}
```

### Scripts NPM Configurados
```json
{
  "scripts": {
    "db:generate": "prisma generate",
    "db:push": "prisma db push",
    "db:migrate": "prisma migrate dev",
    "db:migrate:prod": "prisma migrate deploy",
    "db:studio": "prisma studio",
    "db:seed": "tsx prisma/seed.ts",
    "db:reset": "prisma migrate reset --force",
    "db:status": "prisma migrate status"
  }

```

### Descripción de Scripts
- **`db:generate`**: Genera el cliente Prisma con tipos TypeScript
- **`db:push`**: Sincroniza el esquema sin crear migraciones (desarrollo)
- **`db:migrate`**: Crea y ejecuta migraciones versionadas
- **`db:migrate:prod`**: Ejecuta migraciones en producción
- **`db:studio`**: Abre interfaz visual de administración
- **`db:seed`**: Pobla la base de datos con datos iniciales
- **`db:reset`**: Resetea la base de datos completamente
- **`db:status`**: Muestra el estado de las migraciones

### Health Check Endpoint
```typescript
// src/routes/health.ts
import { Router } from 'express';
import { prisma } from '../lib/prisma';

const router = Router();

router.get('/database', async (req, res) => {
  try {
    // Test básico de conexión
    await prisma.$queryRaw`SELECT 1`;
    
    // Test de performance
    const start = Date.now();
    await prisma.usuario.count();
    const duration = Date.now() - start;
    
    res.json({
      status: 'healthy',
      database: 'postgresql',
      connection: 'active',
      response_time_ms: duration,
      timestamp: new Date().toISOString()
    });

    res.status(503).json({
      status: 'unhealthy',
      database: 'postgresql',
      connection: 'failed',
      error: error.message,
      timestamp: new Date().toISOString()
    });
  }
});

export default router;
```

```typescript
// src/routes/health.ts - endpoint adicional
router.get('/database/stats', async (req, res) => {
  try {
    const stats = await prisma.$queryRaw`
      SELECT 
        schemaname,
        tablename,
        attname,
        n_distinct,
        correlation
      FROM pg_stats 
      WHERE schemaname = 'public'
      ORDER BY tablename, attname;
    `;
    
    const tableStats = await prisma.$queryRaw`
      SELECT 
        table_name,
        pg_size_pretty(pg_total_relation_size(quote_ident(table_name))) as size
      FROM information_schema.tables 
      WHERE table_schema = 'public'
      ORDER BY pg_total_relation_size(quote_ident(table_name)) DESC;
    `;
    
    res.json({
      status: 'success',
      column_stats: stats,
      table_sizes: tableStats,
      timestamp: new Date().toISOString()
    });

      status: 'error',

    });
```

```typescript
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient({
  datasources: {
    db: {
      url: process.env.DATABASE_TEST_URL,
    },
  },
});

beforeAll(async () => {
  await prisma.$connect();
});

afterAll(async () => {
  await prisma.$disconnect();
});

beforeEach(async () => {
  // Limpiar datos entre tests
  const tablenames = await prisma.$queryRaw<Array<{ tablename: string }>>`
    SELECT tablename FROM pg_tables WHERE schemaname='public'
  `;
  
  for (const { tablename } of tablenames) {
    if (tablename !== '_prisma_migrations') {
      await prisma.$executeRawUnsafe(`TRUNCATE TABLE "public"."${tablename}" CASCADE;`);
    }
  }
});
```

## Backup y Recuperación

### Script de Backup Automático
```bash
#!/bin/bash
# scripts/backup-db.sh

DB_NAME="webfestival"
DB_USER="webfestival_user"
BACKUP_DIR="/backups"
DATE=$(date +%Y%m%d_%H%M%S)

# Crear backup
pg_dump -h localhost -U $DB_USER -d $DB_NAME -f "$BACKUP_DIR/webfestival_$DATE.sql"

# Comprimir backup
gzip "$BACKUP_DIR/webfestival_$DATE.sql"

# Limpiar backups antiguos (mantener últimos 7 días)
find $BACKUP_DIR -name "webfestival_*.sql.gz" -mtime +7 -delete

echo "Backup completado: webfestival_$DATE.sql.gz"
```

### Script de Restauración
```bash
#!/bin/bash
# scripts/restore-db.sh

if [ -z "$1" ]; then
  echo "Uso: ./restore-db.sh <archivo_backup>"
  exit 1
fi

DB_NAME="webfestival"
DB_USER="webfestival_user"
BACKUP_FILE="$1"

# Descomprimir si es necesario
if [[ $BACKUP_FILE == *.gz ]]; then
  gunzip -c "$BACKUP_FILE" | psql -h localhost -U $DB_USER -d $DB_NAME
else
  psql -h localhost -U $DB_USER -d $DB_NAME -f "$BACKUP_FILE"
fi

echo "Restauración completada desde: $BACKUP_FILE"
```
