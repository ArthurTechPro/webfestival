### 1. Modelo Principal de Contenido

#### Contenido Base
```prisma
model Contenido {
  id                Int             @id @default(autoincrement())
  titulo            String
  slug              String          @unique
  tipo_contenido    TipoContenido
  estado            EstadoContenido @default(BORRADOR)
  autor_id          String
  fecha_publicacion DateTime?
  created_at        DateTime        @default(now())
  updated_at        DateTime        @updatedAt

  // Relaciones especializadas
  configuracion ContenidoConfiguracion?
  seo           ContenidoSEO?
  metricas      ContenidoMetricas?
  
  // Relaciones de taxonomía
  taxonomias ContenidoTaxonomia[]
  
  // Relaciones de interacción
  comentarios ContenidoComentarios[]
  likes       ContenidoLikes[]
  reportes    ContenidoReportes[]
  
  // Relación con autor
  autor Usuario @relation(fields: [autor_id], references: [id], onDelete: Cascade)

  @@map("contenidos")
}

enum TipoContenido {
  PAGINA_ESTATICA      // Páginas informativas del sitio
  BLOG_POST           // Artículos del blog
  SECCION_CMS         // Secciones editables del CMS
  CONTENIDO_EDUCATIVO // Material educativo y tutoriales
}

enum EstadoContenido {
  BORRADOR    // En edición, no visible públicamente
  REVISION    // Pendiente de aprobación
  PUBLICADO   // Visible públicamente
  ARCHIVADO   // No visible pero conservado
}
```

### Triggers para Métricas Automáticas
```sql
-- Trigger para actualizar contadores de comentarios
CREATE OR REPLACE FUNCTION update_comentarios_count()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        UPDATE contenido_metricas 
        SET total_comentarios = total_comentarios + 1
        WHERE contenido_id = NEW.contenido_id;
        RETURN NEW;
    ELSIF TG_OP = 'DELETE' THEN
        UPDATE contenido_metricas 
        SET total_comentarios = total_comentarios - 1
        WHERE contenido_id = OLD.contenido_id;
        RETURN OLD;
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_update_comentarios_count
    AFTER INSERT OR DELETE ON contenido_comentarios
    FOR EACH ROW EXECUTE FUNCTION update_comentarios_count();

-- Trigger para actualizar contadores de likes
CREATE OR REPLACE FUNCTION update_likes_count()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        UPDATE contenido_metricas

        WHERE contenido_id = NEW.contenido_id;

$$ LANGUAGE plpgsql;

import { z } from 'zod';
import { TipoContenido, EstadoContenido, TipoTaxonomia } from '@prisma/client';

export const CreateContenidoSchema = z.object({
  titulo: z.string().min(5, 'Título debe tener al menos 5 caracteres'),
  slug: z.string().regex(/^[a-z0-9-]+$/, 'Slug debe contener solo letras minúsculas, números y guiones'),
  tipo_contenido: z.nativeEnum(TipoContenido),
  contenido: z.string().min(20, 'Contenido debe tener al menos 20 caracteres'),
  plantilla: z.string().optional(),
  meta_title: z.string().max(60, 'Meta title no debe exceder 60 caracteres').optional(),
  meta_description: z.string().max(160, 'Meta description no debe exceder 160 caracteres').optional(),
  categorias: z.array(z.string()).optional(),
  etiquetas: z.array(z.string()).optional()
});

export const UpdateContenidoSchema = CreateContenidoSchema.partial().extend({
  estado: z.nativeEnum(EstadoContenido).optional()
});

export const CreateComentarioSchema = z.object({

  parent_id: z.number().int().positive().optional()

// prisma/seed-cms.ts
import { PrismaClient, TipoContenido, EstadoContenido, TipoTaxonomia } from '@prisma/client';

const prisma = new PrismaClient();

async function seedCMS() {
  console.log('🌱 Seeding CMS data...');

  // 1. Crear contenido de páginas estáticas
  const paginaInicio = await prisma.contenido.create({
    data: {
      titulo: 'Bienvenido a WebFestival',
      slug: 'inicio',
      tipo_contenido: TipoContenido.PAGINA_ESTATICA,
      estado: EstadoContenido.PUBLICADO,
      autor_id: 'admin-user-id',
      fecha_publicacion: new Date(),
      configuracion: {
        create: {
          contenido: '<h1>Bienvenido a WebFestival</h1><p>La plataforma líder para concursos multimedia creativos.</p>',
          plantilla: 'homepage',
          mostrar_fecha: false,
          mostrar_autor: false,
          permitir_comentarios: false
        }
      },
      seo: {
        create: {
          meta_title: 'WebFestival - Concursos Multimedia Creativos',
          meta_description: 'Participa en concursos de fotografía, video, audio y cortometrajes. Conecta con artistas creativos de todo el mundo.',
          og_title: 'WebFestival - Plataforma de Concursos Creativos',
          og_description: 'La comunidad creativa más grande para concursos multimedia',
          og_image: '/images/og-webfestival.jpg'
        }
      },
      metricas: {
        create: {
          vistas_totales: 0,
          vistas_unicas: 0
        }
      }
    }
  });

  // 2. Crear contenido educativo
  const tutorialFoto = await prisma.contenido.create({
    data: {
      titulo: 'Fundamentos de Composición Fotográfica',
      slug: 'tutorial-composicion-fotografica',
      tipo_contenido: TipoContenido.CONTENIDO_EDUCATIVO,
      estado: EstadoContenido.PUBLICADO,
      autor_id: 'admin-user-id',
      fecha_publicacion: new Date(),
      configuracion: {
        create: {
          contenido: '<h2>Regla de los Tercios</h2><p>La regla de los tercios es fundamental...</p>',
          plantilla: 'tutorial',
          campos_extra: {
            duracion_lectura: 15,
            nivel_dificultad: 'principiante',
            recursos_adicionales: ['video-ejemplo.mp4', 'ejercicios.pdf']
          }
        }
      },
      taxonomias: {
        createMany: {
          data: [
            { tipo: TipoTaxonomia.CATEGORIA, valor: 'Fotografía' },
            { tipo: TipoTaxonomia.ETIQUETA, valor: 'composición' },
            { tipo: TipoTaxonomia.ETIQUETA, valor: 'principiante' },
            { tipo: TipoTaxonomia.NIVEL, valor: 'básico' }
          ]
        }
      }
    }
  });

  // 3. Crear suscriptores de newsletter
  await prisma.newsletterSuscriptor.createMany({
    data: [
      {
        email: 'usuario1@example.com',
        nombre: 'Ana García',
        categorias_interes: ['Fotografía', 'Video']
      },
      {
        email: 'usuario2@example.com',
        nombre: 'Carlos López',
        categorias_interes: ['Audio', 'Cortometrajes']
      }
    ]
  });

  console.log('✅ CMS seeding completado');
}
```
