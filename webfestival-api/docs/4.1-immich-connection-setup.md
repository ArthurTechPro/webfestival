import { z } from 'zod';

const ImmichConfigSchema = z.object({
  IMMICH_SERVER_URL: z.string().url('URL del servidor Immich inválida'),
  IMMICH_API_KEY: z.string().min(20, 'API key de Immich debe tener al menos 20 caracteres'),
  IMMICH_TIMEOUT: z.string().transform(Number).pipe(z.number().min(5000).max(60000)).default('30000'),
  IMMICH_MAX_RETRIES: z.string().transform(Number).pipe(z.number().min(1).max(10)).default('3'),
  IMMICH_RETRY_DELAY: z.string().transform(Number).pipe(z.number().min(500).max(5000)).default('1000')
});

export const immichConfig = ImmichConfigSchema.parse(process.env);

export function validateImmichConfig() {
  try {
    const config = ImmichConfigSchema.parse(process.env);
    console.log('✅ Configuración de Immich validada');
    console.log(`🔗 Servidor: ${config.IMMICH_SERVER_URL}`);
    console.log(`⏱️ Timeout: ${config.IMMICH_TIMEOUT}ms`);
    console.log(`🔄 Max reintentos: ${config.IMMICH_MAX_RETRIES}`);
    return config;
  } catch (error) {
    console.error('❌ Error en configuración de Immich:', error);
    throw new Error('Configuración de Immich inválida');
  }
}

### 2. Servicio de Conexión Immich

#### Servicio Principal
```typescript
// src/services/immich.service.ts
import { init, getServerVersion, getAllAssets } from '@immich/sdk';
import { immichConfig, validateImmichConfig } from '../config/immich';
import axios, { AxiosInstance } from 'axios';

export class ImmichService {
  private isConnected: boolean = false;
  private connectionAttempts: number = 0;
  private lastConnectionAttempt: Date | null = null;
  private axiosInstance: AxiosInstance;

  constructor() {
    // Validar configuración al instanciar
    validateImmichConfig();
    
    // Configurar cliente HTTP con reintentos
    this.axiosInstance = axios.create({
      baseURL: immichConfig.IMMICH_SERVER_URL,
      timeout: immichConfig.IMMICH_TIMEOUT,
      headers: {
        'X-API-Key': immichConfig.IMMICH_API_KEY,
        'Content-Type': 'application/json'
      }
    });

    // Configurar interceptores para reintentos
    this.setupRetryInterceptors();
  }

  /**
   * Inicializar conexión con Immich
   */
  async initialize(): Promise<void> {
    try {
      console.log('🔄 Inicializando conexión con Immich...');
      
      // Inicializar SDK de Immich
      init({
        baseUrl: immichConfig.IMMICH_SERVER_URL,
        apiKey: immichConfig.IMMICH_API_KEY
      });

      // Verificar conexión
      await this.verifyConnection();
      
      this.isConnected = true;
      this.connectionAttempts = 0;
      console.log('✅ Conexión con Immich establecida exitosamente');
      
    } catch (error) {
      this.isConnected = false;
      this.connectionAttempts++;
      this.lastConnectionAttempt = new Date();
      
      console.error('❌ Error conectando con Immich:', error);
      throw new Error(`Fallo al conectar con Immich: ${error.message}`);
    }
  }

  /**
   * Verificar estado de conexión
   */
  async verifyConnection(): Promise<boolean> {
    try {
      const serverInfo = await getServerVersion();
      console.log(`📊 Immich Server Version: ${serverInfo.major}.${serverInfo.minor}.${serverInfo.patch}`);
      
      // Test adicional con endpoint de assets
      await this.axiosInstance.get('/api/assets', {
        params: { take: 1 }
      });
      
      this.isConnected = true;
      return true;
    } catch (error) {
      console.error('❌ Error verificando conexión con Immich:', error);
      this.isConnected = false;
      return false;
    }
  }

  /**
   * Obtener estado de salud de Immich
   */
  async getHealthStatus(): Promise<{
    status: 'healthy' | 'unhealthy';
    connected: boolean;
    server_version?: string;
    last_check: Date;
    connection_attempts: number;
    last_error?: string;
  }> {
    const healthStatus = {
      status: 'unhealthy' as const,
      connected: this.isConnected,
      last_check: new Date(),
      connection_attempts: this.connectionAttempts,
      last_connection_attempt: this.lastConnectionAttempt
    };

    try {
      const isHealthy = await this.verifyConnection();
      
      if (isHealthy) {
        const serverInfo = await getServerVersion();
        return {
          ...healthStatus,
          status: 'healthy',
          connected: true,
          server_version: `${serverInfo.major}.${serverInfo.minor}.${serverInfo.patch}`
        };
      }
    } catch (error) {
      return {
        ...healthStatus,
        last_error: error.message
      };
    }

    return healthStatus;
  }

  /**
   * Asegurar que la conexión esté activa
   */
  ensureConnection(): void {
    if (!this.isConnected) {
      throw new Error('Conexión con Immich no disponible. Verifique la configuración del servidor.');
    }
  }

  /**
   * Reconectar con reintentos
   */
  async reconnect(): Promise<void> {
    const maxRetries = immichConfig.IMMICH_MAX_RETRIES;
    let attempt = 0;

    while (attempt < maxRetries) {
      try {
        await this.initialize();
        return;
      } catch (error) {
        attempt++;
        console.log(`🔄 Intento de reconexión ${attempt}/${maxRetries} falló`);
        
        if (attempt < maxRetries) {
          await this.delay(immichConfig.IMMICH_RETRY_DELAY * attempt);
        }
      }
    }

    throw new Error(`Fallo al reconectar con Immich después de ${maxRetries} intentos`);
  }

  /**
   * Configurar interceptores de reintentos para Axios
   */
  private setupRetryInterceptors(): void {
    this.axiosInstance.interceptors.response.use(
      (response) => response,
      async (error) => {
        const config = error.config;
        
        // No reintentar si ya se agotaron los intentos
        if (!config || config.__retryCount >= immichConfig.IMMICH_MAX_RETRIES) {
          return Promise.reject(error);
        }

        // Incrementar contador de reintentos
        config.__retryCount = config.__retryCount || 0;
        config.__retryCount++;

        // Esperar antes del reintento
        await this.delay(immichConfig.IMMICH_RETRY_DELAY * config.__retryCount);

        console.log(`🔄 Reintentando request a Immich (${config.__retryCount}/${immichConfig.IMMICH_MAX_RETRIES})`);
        
        return this.axiosInstance(config);
      }
    );
  }

  /**
   * Función de delay para reintentos
   */
  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }

  /**
   * Obtener información del servidor
   */
  async getServerInfo() {
    this.ensureConnection();
    
    try {
      const serverVersion = await getServerVersion();
      const response = await this.axiosInstance.get('/api/server-info/stats');
      
      return {
        version: `${serverVersion.major}.${serverVersion.minor}.${serverVersion.patch}`,
        stats: response.data,
        connection_status: 'connected',
        last_check: new Date()
      };
    } catch (error) {
      throw new Error(`Error obteniendo información del servidor: ${error.message}`);
    }
  }
}

export const immichService = new ImmichService();
```

### 3. Middleware de Verificación

#### Middleware para Rutas que Requieren Immich
```typescript
// src/middleware/immich.middleware.ts
import { Request, Response, NextFunction } from 'express';
import { immichService } from '../services/immich.service';

/**
 * Middleware para verificar que Immich esté disponible
 */
export function requireImmichConnection(req: Request, res: Response, next: NextFunction): void {
  try {
    immichService.ensureConnection();
    next();
  } catch (error) {
    res.status(503).json({
      success: false,
      error: 'Servicio de medios no disponible temporalmente',
      code: 'IMMICH_UNAVAILABLE',
      message: 'El servidor de medios está experimentando problemas. Intenta de nuevo más tarde.'
    });
  }
}

/**
 * Inicializar conexión con Immich al arrancar el servidor
 */
export async function initializeImmich(): Promise<void> {
  try {
    await immichService.initialize();
    console.log('🎉 Immich inicializado correctamente');
  } catch (error) {
    console.error('⚠️ Advertencia: No se pudo conectar con Immich al inicio');
    console.error('   El sistema funcionará con funcionalidad limitada');
    console.error('   Error:', error.message);
    
    // No fallar el inicio del servidor, solo advertir
    // El sistema puede funcionar sin Immich para otras funcionalidades
  }
}
```

### 4. Health Checks

#### Endpoint de Health Check
```typescript
// src/routes/health.ts
import { Router } from 'express';
import { immichService } from '../services/immich.service';

const router = Router();

router.get('/immich', async (req, res) => {
  try {
    const healthStatus = await immichService.getHealthStatus();
    
    const statusCode = healthStatus.status === 'healthy' ? 200 : 503;
    
    res.status(statusCode).json({
      service: 'immich',
      ...healthStatus
    });
  } catch (error) {
    res.status(503).json({
      service: 'immich',
      status: 'unhealthy',
      connected: false,
      error: error.message,
      last_check: new Date()
    });
  }
});

router.get('/immich/info', async (req, res) => {
  try {
    const serverInfo = await immichService.getServerInfo();
    
    res.json({
      service: 'immich',
      status: 'healthy',
      ...serverInfo
    });
  } catch (error) {
    res.status(503).json({
      service: 'immich',
      status: 'error',
      error: error.message,
      timestamp: new Date()
    });
  }
});

export default router;
```
