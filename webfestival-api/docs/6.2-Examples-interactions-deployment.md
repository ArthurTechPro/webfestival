### Docker Configuration

#### Dockerfile

```dockerfile
FROM node:18-alpine AS base

# Install dependencies only when needed
FROM base AS deps
RUN apk add --no-cache libc6-compat
WORKDIR /app

COPY package.json package-lock.json* ./
RUN npm ci --only=production

# Rebuild the source code only when needed
FROM base AS builder
WORKDIR /app
COPY --from=deps /app/node_modules ./node_modules
COPY . .

# Generate Prisma client
RUN npx prisma generate

# Build the application
RUN npm run build

# Production image, copy all the files and run the app
FROM base AS runner
WORKDIR /app

ENV NODE_ENV production

RUN addgroup --system --gid 1001 nodejs
RUN adduser --system --uid 1001 webfestival

COPY --from=builder /app/dist ./dist
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/package.json ./package.json
COPY --from=builder /app/prisma ./prisma

USER webfestival

EXPOSE 3001

ENV PORT 3001

CMD ["npm", "start"]
```

    server_name api.webfestival.com;

    ssl_certificate /path/to/ssl/cert.pem;
    ssl_certificate_key /path/to/ssl/key.pem;

    # Rate limiting
    limit_req_zone $binary_remote_addr zone=api:10m rate=10r/s;
    limit_req_zone $binary_remote_addr zone=interactions:10m rate=5r/s;

    # General API routes
    location /api/ {
        limit_req zone=api burst=20 nodelay;
        
        proxy_pass http://webfestival_api;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        
        # Timeouts
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }

    # Stricter rate limiting for interactions
    location /api/v1/interactions/ {
        limit_req zone=interactions burst=10 nodelay;
        
        proxy_pass http://webfestival_api;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }

    # Health check endpoint (no rate limiting)
    location /health {
        proxy_pass http://webfestival_api;
        access_log off;
    }
}

#### Métricas con Prometheus

```javascript
// src/middleware/metrics.ts
import promClient from 'prom-client';

// Crear métricas personalizadas
const httpRequestDuration = new promClient.Histogram({
  name: 'http_request_duration_seconds',
  help: 'Duration of HTTP requests in seconds',
  labelNames: ['method', 'route', 'status_code'],
  buckets: [0.1, 0.5, 1, 2, 5]
});

const interactionCounter = new promClient.Counter({
  name: 'interactions_total',
  help: 'Total number of interactions',
  labelNames: ['type', 'action']
});

const moderationCounter = new promClient.Counter({
  name: 'moderation_actions_total',
  help: 'Total number of moderation actions',
  labelNames: ['action', 'moderator']
});

export const metricsMiddleware = (req: Request, res: Response, next: NextFunction) => {
  const start = Date.now();
  
  res.on('finish', () => {
    const duration = (Date.now() - start) / 1000;
    httpRequestDuration
      .labels(req.method, req.route?.path || req.path, res.statusCode.toString())
      .observe(duration);
  });
  
  next();
};

export { interactionCounter, moderationCounter };
```

### Backup y Recuperación

#### Script de Backup Automático

```bash
#!/bin/bash
# backup-interactions.sh

DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR="/backups/webfestival"
DB_NAME="webfestival"
DB_USER="postgres"
DB_HOST="localhost"

# Crear directorio de backup si no existe
mkdir -p $BACKUP_DIR

# Backup de tablas de interacciones
pg_dump -h $DB_HOST -U $DB_USER -d $DB_NAME \
  --table=contenido_likes \
  --table=contenido_comentarios \
  --table=contenido_reportes \
  --table=contenido_metricas \
  --data-only \
  --file="$BACKUP_DIR/interactions_$DATE.sql"

# Comprimir backup
gzip "$BACKUP_DIR/interactions_$DATE.sql"

# Eliminar backups antiguos (mantener últimos 30 días)
find $BACKUP_DIR -name "interactions_*.sql.gz" -mtime +30 -delete

echo "Backup completado: interactions_$DATE.sql.gz"
```

### Optimizaciones de Rendimiento

#### Configuración de Redis para Caché

```javascript
// src/lib/redis.ts
import Redis from 'ioredis';

const redis = new Redis({
  host: process.env.REDIS_HOST || 'localhost',
  port: parseInt(process.env.REDIS_PORT || '6379'),
  password: process.env.REDIS_PASSWORD,
  retryDelayOnFailover: 100,
  maxRetriesPerRequest: 3,
  lazyConnect: true
});

// Caché para estadísticas (TTL: 5 minutos)
export const cacheStats = async (key: string, data: any) => {
  await redis.setex(`stats:${key}`, 300, JSON.stringify(data));
};

export const getCachedStats = async (key: string) => {
  const cached = await redis.get(`stats:${key}`);
  return cached ? JSON.parse(cached) : null;
};

// Caché para contadores de likes (TTL: 1 hora)
export const cacheLikeCount = async (contentId: number, contentType: string, count: number) => {
  await redis.setex(`likes:${contentType}:${contentId}`, 3600, count.toString());
};
```

### Troubleshooting Común

#### Problemas de Rendimiento

1. **Consultas lentas**: Verificar índices y usar EXPLAIN ANALYZE
2. **Memory leaks**: Monitorear uso de memoria con herramientas como clinic.js
3. **Connection pool exhausted**: Ajustar configuración de pool

#### Problemas de Conectividad

1. **Database timeouts**: Verificar configuración de red y firewall
2. **Redis connection issues**: Verificar configuración y disponibilidad
3. **Load balancer issues**: Verificar health checks y configuración

#### Problemas de Seguridad

1. **Rate limiting bypass**: Verificar configuración de Nginx y aplicación
2. **JWT token issues**: Verificar configuración de secretos y expiración
3. **SQL injection attempts**: Monitorear logs y verificar validación de entrada
