```typescript
// src/services/media.service.ts
import { ImmichApi } from '@immich/sdk';
import { prisma } from '../lib/prisma';
import { immichConfig } from '../config/immich';
import { z } from 'zod';

export interface MediaUploadRequest {
  concursoId: string;
  categoriaId: string;
  titulo: string;
  descripcion?: string;
  tipo: 'FOTOGRAFIA' | 'VIDEO' | 'AUDIO' | 'CORTO_CINE';
}

export interface MediaUploadResponse {
  uploadUrl: string;
  uploadToken: string;
  maxFileSize: number;
  allowedFormats: string[];
  expiresAt: Date;
}

export interface ProcessedMediaResult {
  id: string;
  immichAssetId: string;
  url: string;
  thumbnailUrl?: string;
  metadata: MediaMetadata;
  processingStatus: 'PROCESSING' | 'COMPLETED' | 'FAILED';
}

export interface MediaMetadata {
  filename: string;
  mimeType: string;
  fileSize: number;
  duration?: number; // Para videos y audios
  dimensions?: {
    width: number;
    height: number;
  };
  exifData?: Record<string, any>; // Para imágenes
  audioInfo?: {
    bitrate: number;
    sampleRate: number;
    channels: number;
  };
}

export class MediaService {
  private immichApi: ImmichApi;

  constructor() {
    this.immichApi = new ImmichApi({
      baseUrl: immichConfig.baseUrl,
      apiKey: immichConfig.apiKey
    });
  }

  /**
   * Generar URL de subida segura
   */
  async generateUploadUrl(
    userId: string,
    request: MediaUploadRequest
  ): Promise<MediaUploadResponse> {
    // Validar permisos del usuario
    await this.validateUserPermissions(userId, request.concursoId);

    // Obtener límites del concurso y plan del usuario
    const limits = await this.getUploadLimits(userId, request.concursoId, request.tipo);

    // Generar token de subida temporal
    const uploadToken = await this.generateUploadToken(userId, request);

    // Crear URL de subida segura
    const uploadUrl = await this.immichApi.createUploadUrl({
      albumId: await this.getOrCreateAlbum(request.concursoId, request.categoriaId),
      maxFileSize: limits.maxFileSize,
      allowedMimeTypes: limits.allowedFormats
    });

    return {
      uploadUrl: uploadUrl.url,
      uploadToken,
      maxFileSize: limits.maxFileSize,
      allowedFormats: limits.allowedFormats,
      expiresAt: new Date(Date.now() + 30 * 60 * 1000) // 30 minutos
    };
  }

  /**
   * Procesar medio subido
   */
  async processUploadedMedia(
    uploadToken: string,
    immichAssetId: string
  ): Promise<ProcessedMediaResult> {
    // Validar token de subida
    const uploadData = await this.validateUploadToken(uploadToken);

    // Obtener información del asset desde Immich
    const assetInfo = await this.immichApi.getAssetInfo(immichAssetId);

    // Extraer metadatos específicos por tipo
    const metadata = await this.extractMetadata(assetInfo, uploadData.tipo);

    // Generar versiones optimizadas
    const optimizedVersions = await this.generateOptimizedVersions(
      immichAssetId,
      uploadData.tipo,
      metadata
    );

    // Crear registro en base de datos
    const medio = await prisma.medio.create({
      data: {
        usuario_id: uploadData.userId,
        concurso_id: uploadData.concursoId,
        categoria_id: uploadData.categoriaId,
        titulo: uploadData.titulo,
        descripcion: uploadData.descripcion,
        tipo: uploadData.tipo,
        immich_asset_id: immichAssetId,
        url: assetInfo.webPath,
        thumbnail_url: optimizedVersions.thumbnail,
        metadata: metadata as any,
        estado: 'ACTIVO'
      }
    });

    return {
      id: medio.id,
      immichAssetId,
      url: assetInfo.webPath,
      thumbnailUrl: optimizedVersions.thumbnail,
      metadata,
      processingStatus: 'COMPLETED'
    };
  }

  /**
   * Validar permisos del usuario
   */
  private async validateUserPermissions(userId: string, concursoId: string): Promise<void> {
    // Verificar que el concurso existe y está activo
    const concurso = await prisma.concurso.findUnique({
      where: { id: concursoId },
      include: { inscripciones: { where: { usuario_id: userId } } }
    });

    if (!concurso) {
      throw new Error('Concurso no encontrado');
    }

    if (concurso.estado !== 'ACTIVO') {
      throw new Error('El concurso no está activo para recibir envíos');
    }

    if (new Date() > concurso.fecha_limite) {
      throw new Error('La fecha límite del concurso ha expirado');
    }

    // Verificar inscripción del usuario
    if (concurso.inscripciones.length === 0) {
      throw new Error('Debes estar inscrito en el concurso para enviar medios');
    }

    // Verificar límite de envíos por usuario (máximo 3 por concurso)
    const enviosExistentes = await prisma.medio.count({
      where: {
        usuario_id: userId,
        concurso_id: concursoId,
        estado: 'ACTIVO'
      }
    });

    if (enviosExistentes >= 3) {
      throw new Error('Has alcanzado el límite máximo de 3 envíos por concurso');
    }
  }

  /**
   * Obtener límites de subida
   */
  private async getUploadLimits(
    userId: string,
    concursoId: string,
    tipo: string
  ): Promise<{
    maxFileSize: number;
    allowedFormats: string[];
  }> {
    // Obtener configuración del concurso
    const concurso = await prisma.concurso.findUnique({
      where: { id: concursoId }
    });

    // Obtener plan del usuario
    const userSubscription = await prisma.userSubscription.findFirst({
      where: {
        usuario_id: userId,
        status: 'ACTIVE'
      },
      include: { plan: true }
    });

    // Límites base por tipo de medio
    const baseLimits = {
      FOTOGRAFIA: {
        maxFileSize: 50 * 1024 * 1024, // 50MB
        allowedFormats: ['image/jpeg', 'image/png', 'image/webp']
      },
      VIDEO: {
        maxFileSize: 500 * 1024 * 1024, // 500MB
        allowedFormats: ['video/mp4', 'video/webm']
      },
      AUDIO: {
        maxFileSize: 100 * 1024 * 1024, // 100MB
        allowedFormats: ['audio/mpeg', 'audio/wav', 'audio/flac']
      },
      CORTO_CINE: {
        maxFileSize: 1024 * 1024 * 1024, // 1GB
        allowedFormats: ['video/mp4']
      }
    };

    let limits = baseLimits[tipo as keyof typeof baseLimits];

    // Aplicar límites del plan de suscripción
    if (userSubscription?.plan) {
      const planLimits = userSubscription.plan.features as any;
      if (planLimits.maxFileSize) {
        limits.maxFileSize = Math.min(limits.maxFileSize, planLimits.maxFileSize);
      }
    }

    // Aplicar límites específicos del concurso
    if (concurso?.configuracion) {
      const concursoConfig = concurso.configuracion as any;
      if (concursoConfig.limits?.[tipo]) {
        const concursoLimits = concursoConfig.limits[tipo];
        if (concursoLimits.maxFileSize) {
          limits.maxFileSize = Math.min(limits.maxFileSize, concursoLimits.maxFileSize);
        }
        if (concursoLimits.allowedFormats) {
          limits.allowedFormats = limits.allowedFormats.filter(format =>
            concursoLimits.allowedFormats.includes(format)
          );
        }
      }
    }

    return limits;
  }

  /**
   * Extraer metadatos específicos por tipo
   */
  private async extractMetadata(
    assetInfo: any,
    tipo: string
  ): Promise<MediaMetadata> {
    const baseMetadata: MediaMetadata = {
      filename: assetInfo.originalFileName,
      mimeType: assetInfo.mimeType,
      fileSize: assetInfo.fileSize
    };

    switch (tipo) {
      case 'FOTOGRAFIA':
        return {
          ...baseMetadata,
          dimensions: {
            width: assetInfo.exifInfo?.imageWidth || 0,
            height: assetInfo.exifInfo?.imageHeight || 0
          },
          exifData: assetInfo.exifInfo
        };

      case 'VIDEO':
      case 'CORTO_CINE':
        return {
          ...baseMetadata,
          duration: assetInfo.duration,
          dimensions: {
            width: assetInfo.videoInfo?.width || 0,
            height: assetInfo.videoInfo?.height || 0
          }
        };

      case 'AUDIO':
        return {
          ...baseMetadata,
          duration: assetInfo.duration,
          audioInfo: {
            bitrate: assetInfo.audioInfo?.bitrate || 0,
            sampleRate: assetInfo.audioInfo?.sampleRate || 0,
            channels: assetInfo.audioInfo?.channels || 0
          }
        };

      default:
        return baseMetadata;
    }
  }

  /**
   * Generar versiones optimizadas
   */
  private async generateOptimizedVersions(
    immichAssetId: string,
    tipo: string,
    metadata: MediaMetadata
  ): Promise<{
    thumbnail: string;
    preview?: string;
    optimized?: string;
  }> {
    const versions: any = {};

    // Generar thumbnail para todos los tipos
    versions.thumbnail = await this.immichApi.generateThumbnail(immichAssetId, {
      size: 300,
      quality: 80
    });

    switch (tipo) {
      case 'FOTOGRAFIA':
        // Generar versión optimizada para web
        versions.optimized = await this.immichApi.generateOptimizedImage(immichAssetId, {
          maxWidth: 1920,
          maxHeight: 1080,
          quality: 85,
          format: 'webp'
        });
        break;

      case 'VIDEO':
      case 'CORTO_CINE':
        // Generar preview de video
        versions.preview = await this.immichApi.generateVideoPreview(immichAssetId, {
          duration: 10, // 10 segundos
          quality: 'medium'
        });
        break;

      case 'AUDIO':
        // Generar waveform visual
        versions.waveform = await this.immichApi.generateAudioWaveform(immichAssetId);
        break;
    }

    return versions;
  }

  /**
   * Generar token de subida temporal
   */
  private async generateUploadToken(
    userId: string,
    request: MediaUploadRequest
  ): Promise<string> {
    const tokenData = {
      userId,
      ...request,
      timestamp: Date.now(),
      expiresAt: Date.now() + 30 * 60 * 1000 // 30 minutos
    };

    // En producción, usar JWT o Redis para tokens temporales
    const token = Buffer.from(JSON.stringify(tokenData)).toString('base64');
    
    // Guardar en caché temporal (Redis en producción)
    await this.storeTemporaryToken(token, tokenData);

    return token;
  }

  /**
   * Validar token de subida
   */
  private async validateUploadToken(token: string): Promise<any> {
    try {
      const tokenData = JSON.parse(Buffer.from(token, 'base64').toString());
      
      if (Date.now() > tokenData.expiresAt) {
        throw new Error('Token de subida expirado');
      }

      // Verificar en caché temporal
      const storedData = await this.getTemporaryToken(token);
      if (!storedData) {
        throw new Error('Token de subida inválido');
      }

      return tokenData;
    } catch (error) {
      throw new Error('Token de subida inválido');
    }
  }

  /**
   * Obtener o crear álbum en Immich
   */
  private async getOrCreateAlbum(concursoId: string, categoriaId: string): Promise<string> {
    const albumName = `Concurso-${concursoId}-Categoria-${categoriaId}`;
    
    // Buscar álbum existente
    const existingAlbums = await this.immichApi.getAllAlbums();
    const existingAlbum = existingAlbums.find(album => album.albumName === albumName);

    if (existingAlbum) {
      return existingAlbum.id;
    }

    // Crear nuevo álbum
    const newAlbum = await this.immichApi.createAlbum({
      albumName,
      description: `Medios del concurso ${concursoId} - Categoría ${categoriaId}`
    });

    return newAlbum.id;
  }

  /**
   * Almacenar token temporal (implementar con Redis en producción)
   */
  private async storeTemporaryToken(token: string, data: any): Promise<void> {
    // Implementación temporal en memoria
    // En producción usar Redis con TTL
    global.temporaryTokens = global.temporaryTokens || new Map();
    global.temporaryTokens.set(token, data);
    
    // Limpiar token después de expiración
    setTimeout(() => {
      global.temporaryTokens?.delete(token);
    }, 30 * 60 * 1000);
  }

  /**
   * Obtener token temporal
   */
  private async getTemporaryToken(token: string): Promise<any> {
    // Implementación temporal en memoria
    return global.temporaryTokens?.get(token);
  }
}

export const mediaService = new MediaService();
```

```typescript
// src/controllers/media.controller.ts
import { Response } from 'express';
import { mediaService } from '../services/media.service';
import { AuthenticatedRequest } from '../middleware/auth.middleware';
import { z } from 'zod';

const MediaUploadRequestSchema = z.object({
  concursoId: z.string().uuid('ID de concurso inválido'),
  categoriaId: z.string().uuid('ID de categoría inválido'),
  titulo: z.string().min(1, 'El título es requerido').max(200, 'El título no puede exceder 200 caracteres'),
  descripcion: z.string().max(1000, 'La descripción no puede exceder 1000 caracteres').optional(),
  tipo: z.enum(['FOTOGRAFIA', 'VIDEO', 'AUDIO', 'CORTO_CINE'], {
    errorMap: () => ({ message: 'Tipo de medio inválido' })
  })
});

const ProcessMediaSchema = z.object({
  uploadToken: z.string().min(1, 'Token de subida requerido'),
  immichAssetId: z.string().min(1, 'ID de asset de Immich requerido')
});

export class MediaController {
  /**
   * POST /api/v1/media/upload-url
   * Generar URL de subida segura
   */
  async generateUploadUrl(req: AuthenticatedRequest, res: Response): Promise<void> {
    try {
      const userId = req.user!.userId;
      const validatedData = MediaUploadRequestSchema.parse(req.body);

      const uploadResponse = await mediaService.generateUploadUrl(userId, validatedData);

      res.json({
        success: true,
        message: 'URL de subida generada exitosamente',
        data: uploadResponse
      });
    } catch (error) {
      console.error('Error generando URL de subida:', error);

      if (error instanceof z.ZodError) {
        res.status(400).json({
          success: false,
          message: 'Datos de entrada inválidos',
          errors: error.errors
        });
        return;
      }

      const statusCode = this.getErrorStatusCode(error.message);
      res.status(statusCode).json({
        success: false,
        message: error.message || 'Error interno del servidor'
      });
    }
  }

  /**
   * POST /api/v1/media/process
   * Procesar medio subido
   */
  async processUploadedMedia(req: AuthenticatedRequest, res: Response): Promise<void> {
    try {
      const validatedData = ProcessMediaSchema.parse(req.body);

      const result = await mediaService.processUploadedMedia(
        validatedData.uploadToken,
        validatedData.immichAssetId
      );

      res.json({
        success: true,
        message: 'Medio procesado exitosamente',
        data: result
      });
    } catch (error) {
      console.error('Error procesando medio:', error);

      if (error instanceof z.ZodError) {
        res.status(400).json({
          success: false,
          message: 'Datos de entrada inválidos',
          errors: error.errors
        });
        return;
      }

      const statusCode = this.getErrorStatusCode(error.message);
      res.status(statusCode).json({
        success: false,
        message: error.message || 'Error interno del servidor'
      });
    }
  }

  /**
   * GET /api/v1/media/user/:userId
   * Obtener medios del usuario
   */
  async getUserMedia(req: AuthenticatedRequest, res: Response): Promise<void> {
    try {
      const { userId } = req.params;
      const { concursoId, tipo, page = '1', limit = '10' } = req.query;

      // Verificar permisos (solo el propio usuario o admin)
      if (req.user!.userId !== userId && req.user!.role !== 'ADMIN') {
        res.status(403).json({
          success: false,
          message: 'No tienes permisos para ver estos medios'
        });
        return;
      }

      const filters: any = {
        usuario_id: userId,
        estado: 'ACTIVO'
      };

      if (concursoId) filters.concurso_id = concursoId as string;
      if (tipo) filters.tipo = tipo as string;

      const pageNum = parseInt(page as string);
      const limitNum = parseInt(limit as string);
      const skip = (pageNum - 1) * limitNum;

      const [medios, total] = await Promise.all([
        prisma.medio.findMany({
          where: filters,
          include: {
            concurso: { select: { titulo: true, estado: true } },
            categoria: { select: { nombre: true } },
            usuario: { select: { nombre: true } }
          },
          orderBy: { created_at: 'desc' },
          skip,
          take: limitNum
        }),
        prisma.medio.count({ where: filters })
      ]);

      res.json({
        success: true,
        data: {
          medios,
          pagination: {
            page: pageNum,
            limit: limitNum,
            total,
            pages: Math.ceil(total / limitNum)
          }
        }
      });
    } catch (error) {
      console.error('Error obteniendo medios del usuario:', error);
      res.status(500).json({
        success: false,
        message: 'Error interno del servidor'
      });
    }
  }

  /**
   * DELETE /api/v1/media/:id
   * Eliminar medio
   */
  async deleteMedia(req: AuthenticatedRequest, res: Response): Promise<void> {
    try {
      const { id } = req.params;
      const userId = req.user!.userId;
      const userRole = req.user!.role;

      // Buscar el medio
      const medio = await prisma.medio.findUnique({
        where: { id },
        include: { concurso: true }
      });

      if (!medio) {
        res.status(404).json({
          success: false,
          message: 'Medio no encontrado'
        });
        return;
      }

      // Verificar permisos (propietario o admin)
      if (medio.usuario_id !== userId && userRole !== 'ADMIN') {
        res.status(403).json({
          success: false,
          message: 'No tienes permisos para eliminar este medio'
        });
        return;
      }

      // Verificar que el concurso permita eliminación
      if (medio.concurso.estado === 'FINALIZADO') {
        res.status(400).json({
          success: false,
          message: 'No se pueden eliminar medios de concursos finalizados'
        });
        return;
      }

      // Marcar como eliminado (soft delete)
      await prisma.medio.update({
        where: { id },
        data: { estado: 'ELIMINADO' }
      });

      res.json({
        success: true,
        message: 'Medio eliminado exitosamente'
      });
    } catch (error) {
      console.error('Error eliminando medio:', error);
      res.status(500).json({
        success: false,
        message: 'Error interno del servidor'
      });
    }
  }

  /**
   * Determinar código de estado HTTP basado en el mensaje de error
   */
  private getErrorStatusCode(message: string): number {
    if (message.includes('no encontrado')) return 404;
    if (message.includes('no está activo') || message.includes('expirado')) return 400;
    if (message.includes('no está inscrito') || message.includes('límite máximo')) return 403;
    if (message.includes('Token') && message.includes('inválido')) return 401;
    return 500;
  }
}

export const mediaController = new MediaController();
```

```typescript
);

  requireRole(['PARTICIPANTE', 'ADMIN']),

);
```
