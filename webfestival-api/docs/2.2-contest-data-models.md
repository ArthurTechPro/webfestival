## Modelos Implementados

### 1. Sistema de Usuarios

#### Modelo Usuario Principal
```prisma
model Usuario {
  id          String   @id @default(cuid())
  email       String   @unique
  password    String?  // Null para OAuth
  nombre      String?
  role        Role     @default(PARTICIPANTE)
  picture_url String?
  bio         String?
  created_at  DateTime @default(now())
  updated_at  DateTime @updatedAt

  // Relaciones de concursos
  inscripciones           Inscripcion[]
  medios                  Medio[]
  calificaciones_jurado   Calificacion[]

  // Relaciones sociales
  seguimientos_seguidor   Seguimiento[] @relation("SeguidorSeguimientos")
  seguimientos_seguido    Seguimiento[] @relation("SeguidoSeguimientos")

  // Relaciones de jurado
  jurado_asignaciones     JuradoAsignacion[]
  jurado_especializaciones JuradoEspecializacion[]

  // Relaciones de contenido
  contenidos_autor        Contenido[]
  
  // Relaciones de suscripción
  suscripcion             UserSubscription?

  @@map("usuarios")
}

enum Role {
  PARTICIPANTE    // Puede participar en concursos
  JURADO         // Puede evaluar medios asignados
  ADMIN          // Administrador completo del sistema
  CONTENT_ADMIN  // Administrador de contenido CMS únicamente
}
```

#### Especialización de Jurados
```prisma
model JuradoEspecializacion {
  id                Int       @id @default(autoincrement())
  usuario_id        String
  especializacion   TipoMedio
  experiencia_años  Int?
  certificaciones   String[]  // Array de certificaciones
  portfolio_url     String?
  
  // Relaciones
  usuario Usuario @relation(fields: [usuario_id], references: [id], onDelete: Cascade)

  @@unique([usuario_id, especializacion])
  @@map("jurado_especializaciones")
}
```

### 2. Sistema de Concursos

#### Modelo Concurso
```prisma
model Concurso {
  id             Int            @id @default(autoincrement())
  titulo         String
  descripcion    String
  reglas         String?
  fecha_inicio   DateTime
  fecha_final    DateTime
  status         StatusConcurso @default(PROXIMAMENTE)
  imagen_url     String?
  max_envios     Int            @default(3)
  tamano_max_mb  Int            @default(10)
  created_at     DateTime       @default(now())

  // Relaciones
  categorias    Categoria[]
  inscripciones Inscripcion[]
  medios        Medio[]

  @@map("concursos")
}

enum StatusConcurso {
  PROXIMAMENTE  // Concurso anunciado pero no iniciado
  ACTIVO       // Período de envío de medios
  CALIFICACION // Período de evaluación por jurados
  FINALIZADO   // Concurso completado con resultados
}
```

#### Modelo Categoría
```prisma
model Categoria {
  id          Int @id @default(autoincrement())
  nombre      String
  concurso_id Int

  // Relaciones
  concurso            Concurso           @relation(fields: [concurso_id], references: [id], onDelete: Cascade)
  medios              Medio[]
  jurado_asignaciones JuradoAsignacion[]

  @@map("categorias")
}
```

#### Modelo Inscripción
```prisma
model Inscripcion {
  id                Int      @id @default(autoincrement())
  usuario_id        String
  concurso_id       Int
  fecha_inscripcion DateTime @default(now())

  // Relaciones
  usuario  Usuario  @relation(fields: [usuario_id], references: [id], onDelete: Cascade)
  concurso Concurso @relation(fields: [concurso_id], references: [id], onDelete: Cascade)

  @@unique([usuario_id, concurso_id])
  @@map("inscripciones")
}
```

### 3. Sistema Multimedia

#### Modelo Medio
```prisma
model Medio {
  id             Int       @id @default(autoincrement())
  titulo         String
  tipo_medio     TipoMedio
  usuario_id     String
  concurso_id    Int
  categoria_id   Int
  medio_url      String
  thumbnail_url  String?
  preview_url    String?
  duracion       Int?      // Para videos y audios (segundos)
  formato        String
  tamano_archivo BigInt
  metadatos      Json?     // EXIF para imágenes, metadata específico por tipo
  fecha_subida   DateTime  @default(now())

  // Relaciones
  usuario        Usuario        @relation(fields: [usuario_id], references: [id], onDelete: Cascade)
  concurso       Concurso       @relation(fields: [concurso_id], references: [id], onDelete: Cascade)
  categoria      Categoria      @relation(fields: [categoria_id], references: [id], onDelete: Cascade)
  calificaciones Calificacion[]
  comentarios    Comentario[]

  @@map("medios")
}

enum TipoMedio {
  fotografia    // Imágenes estáticas (JPEG, PNG, WebP)
  video        // Videos cortos (MP4, WebM, MOV)
  audio        // Archivos de audio (MP3, WAV, FLAC)
  corto_cine   // Cortometrajes (MP4, WebM, MOV con mayor duración)
}
```

### 4. Sistema de Evaluación

#### Modelo Criterio Dinámico
```prisma
model Criterio {
  id          Int        @id @default(autoincrement())
  nombre      String     @unique
  descripcion String?
  tipo_medio  TipoMedio? // null = criterio universal
  peso        Decimal    @default(1.0) @db.Decimal(5, 2)
  activo      Boolean    @default(true)
  orden       Int        @default(0)

  // Relaciones
  calificaciones_detalle CalificacionDetalle[]

  @@map("criterios")
}
```

#### Modelo Calificación
```prisma
model Calificacion {
  id                  Int      @id @default(autoincrement())
  medio_id            Int
  jurado_id           String
  comentarios         String?
  fecha_calificacion  DateTime @default(now())

  // Relaciones
  medio    Medio                 @relation(fields: [medio_id], references: [id], onDelete: Cascade)
  jurado   Usuario               @relation(fields: [jurado_id], references: [id], onDelete: Cascade)
  detalles CalificacionDetalle[]

  @@unique([medio_id, jurado_id])
  @@map("calificaciones")
}

model CalificacionDetalle {
  id              Int @id @default(autoincrement())
  calificacion_id Int
  criterio_id     Int
  puntuacion      Int @db.SmallInt // 1-10

  // Relaciones
  calificacion Calificacion @relation(fields: [calificacion_id], references: [id], onDelete: Cascade)
  criterio     Criterio     @relation(fields: [criterio_id], references: [id], onDelete: Cascade)

  @@unique([calificacion_id, criterio_id])
  @@map("calificaciones_detalle")
}
```

#### Modelo Asignación de Jurados

```prisma
model Seguimiento {
```

```sql
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
```

```typescript
import { PrismaClient, Role, TipoMedio, StatusConcurso } from '@prisma/client';
import bcrypt from 'bcryptjs';

const prisma = new PrismaClient();

async function main() {

      nombre: 'María González',
      role: Role.JURADO,
      bio: 'Fotógrafa profesional con 15 años de experiencia'
    }
  });

  const participante = await prisma.usuario.create({
    data: {
      email: 'participante@webfestival.com',
      password: await bcrypt.hash('participante123', 10),
      nombre: 'Carlos Rodríguez',
      role: Role.PARTICIPANTE,
      bio: 'Fotógrafo aficionado apasionado por la naturaleza'
    }
  });

  // 2. Crear especialización de jurado
  await prisma.juradoEspecializacion.create({
    data: {
      usuario_id: juradoFoto.id,
      especializacion: TipoMedio.fotografia,
      experiencia_años: 15,
      certificaciones: ['Certificación Nacional de Fotografía', 'Master en Fotografía Digital'],
      portfolio_url: 'https://mariagonzalez-portfolio.com'
    }
  });

  // 3. Crear concurso de ejemplo
  const concurso = await prisma.concurso.create({
    data: {
      titulo: 'Festival de Fotografía Urbana 2024',
      descripcion: 'Concurso de fotografía que celebra la belleza y diversidad de la vida urbana',
      reglas: 'Las fotografías deben ser originales y capturar aspectos de la vida urbana contemporánea.',
      fecha_inicio: new Date('2024-01-15'),
      fecha_final: new Date('2024-03-15'),
      status: StatusConcurso.ACTIVO,
      max_envios: 3,
      tamano_max_mb: 10
    }
  });

  // 4. Crear categorías
  const categoriaRetrato = await prisma.categoria.create({
    data: {
      nombre: 'Retrato Urbano',
      concurso_id: concurso.id
    }
  });

  const categoriaArquitectura = await prisma.categoria.create({
    data: {
      nombre: 'Arquitectura Moderna',
      concurso_id: concurso.id
    }
  });

  // 5. Crear inscripción
  await prisma.inscripcion.create({
    data: {
      usuario_id: participante.id,
      concurso_id: concurso.id
    }
  });

  // 6. Crear asignación de jurado
  await prisma.juradoAsignacion.createMany({
    data: [
      {
        usuario_id: juradoFoto.id,
        categoria_id: categoriaRetrato.id
      },
      {
        usuario_id: juradoFoto.id,
        categoria_id: categoriaArquitectura.id
      }
    ]
  });

  console.log('✅ Seeding completado exitosamente');
  console.log(`👤 Admin creado: ${adminUser.email}`);
  console.log(`👨‍⚖️ Jurado creado: ${juradoFoto.email}`);
  console.log(`🎨 Participante creado: ${participante.email}`);
  console.log(`🏆 Concurso creado: ${concurso.titulo}`);
}

main()
  .catch((e) => {
    console.error('❌ Error durante el seeding:', e);
    process.exit(1);
  })
  .finally(async () => {
    await prisma.$disconnect();
  });
```

## Validaciones de Modelo

### Validaciones con Zod
```typescript
// src/schemas/user.schemas.ts
import { z } from 'zod';
import { Role, TipoMedio } from '@prisma/client';

export const CreateUserSchema = z.object({
  email: z.string().email('Email inválido'),
  password: z.string().min(8, 'Password debe tener al menos 8 caracteres'),
  nombre: z.string().min(2, 'Nombre debe tener al menos 2 caracteres'),
  role: z.nativeEnum(Role).default(Role.PARTICIPANTE),
  bio: z.string().max(500, 'Bio no puede exceder 500 caracteres').optional(),
  picture_url: z.string().url('URL de imagen inválida').optional()
});

export const CreateConcursoSchema = z.object({
  titulo: z.string().min(5, 'Título debe tener al menos 5 caracteres'),
  descripcion: z.string().min(20, 'Descripción debe tener al menos 20 caracteres'),
  reglas: z.string().optional(),
  fecha_inicio: z.string().datetime('Fecha de inicio inválida'),
  fecha_final: z.string().datetime('Fecha final inválida'),
  max_envios: z.number().int().min(1).max(10).default(3),
  tamano_max_mb: z.number().int().min(1).max(100).default(10)
}).refine((data) => {
  return new Date(data.fecha_final) > new Date(data.fecha_inicio);
}, {
  message: 'Fecha final debe ser posterior a fecha de inicio',
  path: ['fecha_final']
});

export const CreateMedioSchema = z.object({
  titulo: z.string().min(3, 'Título debe tener al menos 3 caracteres'),
  tipo_medio: z.nativeEnum(TipoMedio),
  categoria_id: z.number().int().positive('ID de categoría inválido'),
  formato: z.string().min(3, 'Formato requerido'),
  tamano_archivo: z.number().positive('Tamaño de archivo inválido')
});
```
